//7. Write a C program containing declaration of three variables (of type int, char and
//float), also assign some values to them and print values of all three variables using
//single printf().

#include<stdio.h>
#include<conio.h>
int main()
{
    int a=5;
    char m=97;
    float f=4.5;
    printf("a=%d m=%c f=%f",a,m,f);
    getch();
}
